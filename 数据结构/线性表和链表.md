5. 设计算法：从顺序表中删除所有值为 x 的元素。
    
    c
    
    Void delall(SqList &L) {
        int i = 0, j;
        while (i < L.length && L.elem[i] != x) i++;
        for (j = i + 1; j < L.length; j++)
            if (L.elem[j] != x) {
                L.elem[i] = L.elem[j];
                i++;
            }
        L.length = i;
    }
