### 📂 实验二：进程的控制（设计 1 + 设计 2）

实验二郝老师大纲要求实现多节点进程树以及 `execve` 程序替换，因此包含两个独立的源文件：

#### 1. 复杂进程树构建

- **文件名**：`tree_fork.c`
    
- **编译命令**：`gcc tree_fork.c -o tree_fork`
    
- **运行命令**：`./tree_fork`
    

```c
#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <stdlib.h>

int main() {
    pid_t pid_b, pid_c, pid_d, pid_e, pid_f;
    printf("[进程 Pa] 启动，PID: %d\n", getpid());

    // 1. Pa 创建第一个子进程 Pb
    pid_b = fork();
    if (pid_b == 0) {
        printf("  [子进程 Pb] 创建成功，PID: %d, 父进程 Pa PID: %d\n", getpid(), getppid());
        // Pb 创建孙进程 Pd
        pid_d = fork();
        if (pid_d == 0) {
            printf("    [孙进程 Pd] 创建成功，PID: %d, 父进程 Pb PID: %d\n", getpid(), getppid());
            exit(0);
        }
        // Pb 创建孙进程 Pe
        pid_e = fork();
        if (pid_e == 0) {
            printf("    [孙进程 Pe] 创建成功，PID: %d, 父进程 Pb PID: %d\n", getpid(), getppid());
            exit(0);
        }
        wait(NULL); wait(NULL);
        exit(0);
    }

    // 2. Pa 创建第二个子进程 Pc
    pid_c = fork();
    if (pid_c == 0) {
        printf("  [子进程 Pc] 创建成功，PID: %d, 父进程 Pa PID: %d\n", getpid(), getppid());
        // Pc 创建孙进程 Pf
        pid_f = fork();
        if (pid_f == 0) {
            printf("    [孙进程 Pf] 创建成功，PID: %d, 父进程 Pc PID: %d\n", getpid(), getppid());
            exit(0);
        }
        wait(NULL);
        exit(0);
    }

    // 3. 父进程 Pa 打印验证信息
    printf("[进程 Pa] 验证 fork 返回值：创建 Pb 的返回值为 %d, 创建 Pc 的返回值为 %d\n", pid_b, pid_c);
    wait(NULL); wait(NULL);
    printf("[进程 Pa] 所有子孙进程已退出，主进程结束。\n");
    return 0;
}
```

### 📂 实验三：进程间的通信（管道、消息队列、共享内存）

实验三要求实现三种底层的 IPC 机制，分为三个独立的源文件：

#### 1. 管道通信

- **文件名**：`pipe_lab.c`
    
- **编译命令**：`gcc pipe_lab.c -o pipe_lab`
    
- **运行命令**：`./pipe_lab`
    
```c
#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    int fd[2];
    pid_t pid;
    char write_msg[] = "Hello, this is a message via Pipe from JNU!";
    char read_msg[100];

    if (pipe(fd) == -1) {
        perror("Pipe 创建失败");
        return 1;
    }

    pid = fork();
    if (pid < 0) {
        perror("Fork 失败");
        return 1;
    } 
    else if (pid == 0) { 
        close(fd[1]); // 关闭写端
        read(fd[0], read_msg, sizeof(read_msg));
        printf("[子进程读端] 成功从管道接收到数据: \"%s\"\n", read_msg);
        close(fd[0]);
    } 
    else { 
        close(fd[0]); // 关闭读端
        printf("[父进程写端] 正在向管道写入数据...\n");
        write(fd[1], write_msg, strlen(write_msg) + 1);
        close(fd[1]);
        wait(NULL); 
    }
    return 0;
}
```

#### 2. 消息队列

- **文件名**：`msg_lab.c`
    
- **编译命令**：`gcc msg_lab.c -o msg_lab`
    
- **运行命令**：`./msg_lab`
    
```c
#include <stdio.h>
#include <sys/ipc.h>
#include <sys/msg.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>

struct msg_buf {
    long msg_type;
    char msg_text[100];
};

int main() {
    key_t key = ftok("msg_key_file", 65); 
    int msgid = msgget(key, 0666 | IPC_CREAT); 
    
    pid_t pid = fork();
    if (pid == 0) { 
        struct msg_buf message;
        msgrcv(msgid, &message, sizeof(message.msg_text), 101, 0);
        printf("[子进程接收端] 成功捕获 Type=101 的特定消息: \"%s\"\n", message.msg_text);
        
        msgrcv(msgid, &message, sizeof(message.msg_text), 102, 0);
        printf("[子进程接收端] 成功捕获 Type=102 的特定消息: \"%s\"\n", message.msg_text);
    } 
    else { 
        struct msg_buf msg1 = {101, "Message Type 101 [High Priority]"};
        struct msg_buf msg2 = {102, "Message Type 102 [Normal Priority]"};
        
        sleep(1); 
        printf("[父进程发送端] 正在投递不同类型的消息至内核队列...\n");
        msgsnd(msgid, &msg1, strlen(msg1.msg_text) + 1, 0);
        msgsnd(msgid, &msg2, strlen(msg2.msg_text) + 1, 0);
        
        wait(NULL);
        msgctl(msgid, IPC_RMID, NULL); // 回收队列资源
        printf("[父进程] 内核消息队列已安全销毁。\n");
    }
    return 0;
}
```

#### 3. 共享内存

- **文件名**：`shm_lab.c`
    
- **编译命令**：`gcc shm_lab.c -o shm_lab`
    
- **运行命令**：`./shm_lab`
    
```c
#include <stdio.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <string.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    key_t key = ftok("shm_key_file", 75);
    int shmid = shmget(key, 1024, 0666 | IPC_CREAT);

    pid_t pid = fork();
    if (pid == 0) { 
        sleep(1); 
        char *str = (char*) shmat(shmid, (void*)0, 0); 
        printf("[子进程读取端] 发现共享区数据已被修改！读取内容: \"%s\"\n", str);
        shmdt(str); 
    } 
    else { 
        char *str = (char*) shmat(shmid, (void*)0, 0); 
        printf("[父进程写入端] 正在直接操作共享区地址空间指针...\n");
        strcpy(str, "This data is placed directly inside Shared Memory!");
        
        wait(NULL);
        shmdt(str); 
        shmctl(shmid, IPC_RMID, NULL); // 回收内核共享内存对象
        printf("[父进程] 内核物理共享内存区已成功释放。\n");
    }
    return 0;
}
```

### 📂 实验四：生产者消费者问题

加入了大纲规定的“严禁连续读写同一单元相同字符”条件，并配合信号量。

- **文件名**：`pc_lab.c`
    
- **编译命令**：`gcc -pthread pc_lab.c -o pc_lab` （⚠️ **注意：必须加 -pthread 库**）
    
- **运行命令**：`./pc_lab`
    
```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <semaphore.h>
#include <fcntl.h>
#include <sys/wait.h>
#include <time.h>

#define N 5 
#define SHM_KEY 0x123456

struct shared_data {
    char buffer[N];
    char last_prod[N]; // 记录各个单元上一次写过的字符
    char last_cons[N]; // 记录各个单元上一次读过的字符
    int in;
    int out;
};

int main() {
    int shmid = shmget(SHM_KEY, sizeof(struct shared_data), 0666 | IPC_CREAT);
    struct shared_data *sd = (struct shared_data *)shmat(shmid, NULL, 0);
    sd->in = 0; sd->out = 0;
    for(int i=0; i<N; i++) {
        sd->buffer[i] = ' ';
        sd->last_prod[i] = '\0';
        sd->last_cons[i] = '\0';
    }

    sem_unlink("/sem_mutex");
    sem_unlink("/sem_empty");
    sem_unlink("/sem_full");
    
    sem_t *mutex = sem_open("/sem_mutex", O_CREAT, 0666, 1);
    sem_t *empty = sem_open("/sem_empty", O_CREAT, 0666, N);
    sem_t *full  = sem_open("/sem_full",  O_CREAT, 0666, 0);

    printf("=== 生产者消费者实验启动（缓冲区大小: %d 字节）===\n", N);

    pid_t pid_prod = fork();
    if (pid_prod == 0) { 
        srand(time(NULL) ^ getpid());
        char items[2] = {'A', 'B'}; 
        int count = 0;

        while (count < 10) { 
            char item = items[rand() % 2]; 
            sem_wait(empty); 
            sem_wait(mutex); 

            int curr_in = sd->in;
            // ⭐ 大纲特殊限制判定：如果该单元上一次写过相同字符，则强制取反
            if (sd->last_prod[curr_in] == item) {
                item = (item == 'A') ? 'B' : 'A';
            }

            sd->buffer[curr_in] = item;
            sd->last_prod[curr_in] = item; 
            printf("[生产者] 正在向单元 [%d] 写入字符: '%c'\n", curr_in, item);
            sd->in = (sd->in + 1) % N; 

            sem_post(mutex); 
            sem_post(full);  
            count++;
            usleep(rand() % 200000); 
        }
        shmdt(sd); exit(0);
    }

    pid_t pid_cons = fork();
    if (pid_cons == 0) { 
        srand(time(NULL) ^ getpid());
        int count = 0;

        while (count < 10) { 
            sem_wait(full);  
            sem_wait(mutex); 

            int curr_out = sd->out;
            char item = sd->buffer[curr_out];

            // ⭐ 大纲特殊限制判定：检查是否连续读取相同字符
            if (sd->last_cons[curr_out] == item) {
                printf("[消费者警告] 检测到单元 [%d] 存在连续相同字符 '%c' 的读取风险！\n", curr_out, item);
            }

            sd->last_cons[curr_out] = item; 
            printf("[消费者] 成功从单元 [%d] 读取字符: '%c'\n", curr_out, item);
            sd->buffer[curr_out] = ' '; 
            sd->out = (sd->out + 1) % N; 

            sem_post(mutex); 
            sem_post(empty); 
            count++;
            usleep(rand() % 300000);
        }
        shmdt(sd); exit(0);
    }

    wait(NULL); wait(NULL);
    sem_close(mutex); sem_close(empty); sem_close(full);
    sem_unlink("/sem_mutex"); sem_unlink("/sem_empty"); sem_unlink("/sem_full");
    shmdt(sd); shmctl(shmid, IPC_RMID, NULL);
    printf("=== 生产者消费者实验安全结束，资源已全部回收 ===\n");
    return 0;
}
```

### 📂 实验五：简单文件系统实现

实现了常驻内存的磁盘与断电序列化持久保存功能。

- **文件名**：`fs_lab.c`
    
- **编译命令**：`gcc fs_lab.c -o fs_lab`
    
- **运行命令**：`./fs_lab`
    
```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_FILES 32
#define MAX_NAME_LEN 20
#define DISK_FILE "vdisk.bin"

typedef struct {
    char name[MAX_NAME_LEN];
    int is_dir;  // 1: 目录, 0: 文件
    int size;    
    int is_used; // 1: 占用, 0: 空闲
} FCB;

typedef struct {
    FCB entries[MAX_FILES];
} VirtualDisk;

VirtualDisk my_disk;

void do_format() {
    for (int i = 0; i < MAX_FILES; i++) {
        my_disk.entries[i].is_used = 0;
        strcpy(my_disk.entries[i].name, "");
    }
    printf("[文件系统] 虚拟磁盘格式化成功，根目录已初始化。\n");
}

void do_ls() {
    printf("\n--- 根目录内容列表 ---\n");
    printf("%-20s %-10s %-10s\n", "名称", "类型", "大小(Bytes)");
    int empty = 1;
    for (int i = 0; i < MAX_FILES; i++) {
        if (my_disk.entries[i].is_used) {
            empty = 0;
            printf("%-20s %-10s %-10d\n", 
                   my_disk.entries[i].name, 
                   my_disk.entries[i].is_dir ? "DIR目录" : "FILE文件", 
                   my_disk.entries[i].size);
        }
    }
    if (empty) printf("(目录为空)\n");
    printf("----------------------\n\n");
}

void do_create_entry(char *name, int is_dir) {
    for (int i = 0; i < MAX_FILES; i++) {
        if (my_disk.entries[i].is_used && strcmp(my_disk.entries[i].name, name) == 0) {
            printf("[错误] 创建失败：名称 \"%s\" 已存在！\n", name);
            return;
        }
    }
    for (int i = 0; i < MAX_FILES; i++) {
        if (!my_disk.entries[i].is_used) {
            strcpy(my_disk.entries[i].name, name);
            my_disk.entries[i].is_dir = is_dir;
            my_disk.entries[i].size = is_dir ? 0 : (rand() % 1024 + 1); 
            my_disk.entries[i].is_used = 1;
            printf("[成功] 已创建 %s: \"%s\"\n", is_dir ? "目录" : "文件", name);
            return;
        }
    }
    printf("[错误] 磁盘目录区已满！\n");
}

void do_delete_entry(char *name, int is_dir) {
    for (int i = 0; i < MAX_FILES; i++) {
        if (my_disk.entries[i].is_used && strcmp(my_disk.entries[i].name, name) == 0) {
            if (my_disk.entries[i].is_dir != is_dir) {
                printf("[错误] 类型不匹配，无法删除！\n");
                return;
            }
            my_disk.entries[i].is_used = 0;
            printf("[成功] 已删除 %s: \"%s\"\n", is_dir ? "目录" : "文件", name);
            return;
        }
    }
    printf("[错误] 未找到指定的名称 \"%s\"\n", name);
}

int main() {
    char cmd[20], arg[MAX_NAME_LEN];
    FILE *fp = fopen(DISK_FILE, "rb");
    if (fp) {
        fread(&my_disk, sizeof(VirtualDisk), 1, fp);
        fclose(fp);
        printf("[系统提示] 成功从 %s 中恢复上一次的文件系统环境。\n", DISK_FILE);
    } else {
        printf("[系统提示] 未发现现存的虚拟磁盘文件，自动执行加载初始化。\n");
        do_format();
    }

    while (1) {
        printf("JNU-OS-FS_Sim> ");
        if (scanf("%s", cmd) == EOF) break;

        if (strcmp(cmd, "exit") == 0) {
            fp = fopen(DISK_FILE, "wb");
            if (fp) {
                fwrite(&my_disk, sizeof(VirtualDisk), 1, fp);
                fclose(fp);
                printf("[文件系统] 现场已安全持久化保存至真实磁盘 %s。退出程序。\n", DISK_FILE);
            }
            break;
        } 
        else if (strcmp(cmd, "format") == 0) { do_format(); } 
        else if (strcmp(cmd, "ls") == 0) { do_ls(); } 
        else if (strcmp(cmd, "mkdir") == 0) { scanf("%s", arg); do_create_entry(arg, 1); } 
        else if (strcmp(cmd, "create") == 0) { scanf("%s", arg); do_create_entry(arg, 0); } 
        else if (strcmp(cmd, "rmdir") == 0) { scanf("%s", arg); do_delete_entry(arg, 1); } 
        else if (strcmp(cmd, "delete") == 0) { scanf("%s", arg); do_delete_entry(arg, 0); } 
        else { printf("[错误] 未知命令！\n"); }
    }
    return 0;
}
```
